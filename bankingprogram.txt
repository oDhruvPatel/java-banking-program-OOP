import java.util.Scanner;

// ============================================================
// MAIN CLASS — handles user interaction only (no business logic)
// This separation is called "Separation of Concerns"
// ============================================================
class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in); // Scanner reads user input from keyboard
        boolean isRunning = true; // controls the while loop
        int choice = 0;

        // --- ACCOUNT CREATION (before the menu loop) ---
        System.out.print("Enter your name: ");
        String name = scanner.nextLine(); // nextLine() reads a full line including spaces

        System.out.print("Enter initial balance: ");
        double initialBalance = scanner.nextDouble();
        scanner.nextLine(); // IMPORTANT: consumes leftover newline character after nextDouble()
                            // Without this, the next nextLine() would read an empty string

        // Creating an OBJECT (instance) of BankAccount class
        // 'new' keyword allocates memory and calls the constructor
        BankAccount account = new BankAccount(name, initialBalance);
        System.out.println("Account created! Welcome, " + name + "!\n");

        // --- MAIN MENU LOOP ---
        while (isRunning) {
            // Display menu options
            System.out.println("Banking Simulation");
            System.out.println();
            System.out.println("1. Deposit");
            System.out.println("2. Withdraw");
            System.out.println("3. Display");
            System.out.println("4. Open a new account");
            System.out.println("5. Exit");
            System.out.println();
            System.out.print("Enter choice 1-5: ");
            choice = scanner.nextInt();
            scanner.nextLine(); // consume leftover newline after nextInt()

            // Switch handles each menu option
            switch (choice) {

                case 1: // DEPOSIT
                    System.out.print("Enter amount to deposit: ");
                    double deposit = scanner.nextDouble();
                    scanner.nextLine();
                    boolean res = account.deposit(deposit); // deposit() returns true/false
                    if (res) System.out.println("Deposited: " + deposit);
                    break; // ALWAYS break — without it, falls through to next case

                case 2: // WITHDRAW
                    System.out.print("Enter amount to withdraw: ");
                    double wit = scanner.nextDouble();
                    scanner.nextLine();
                    boolean withdrawn = account.withdraw(wit); // withdraw() returns true/false
                    if (withdrawn) {
                        double remaining = account.showBalance(); // showBalance() returns balance AND prints it
                        System.out.println("Withdrawn: " + wit + " | Remaining: " + remaining);
                    }
                    break;

                case 3: // DISPLAY BALANCE
                    account.showBalance(); // calling method on the object
                    break; // don't forget break — missing this falls through to case 4/5

                case 4: // OPEN NEW ACCOUNT
                    System.out.print("Enter new account holder name: ");
                    String newName = scanner.nextLine();
                    System.out.print("Enter initial balance: ");
                    double newBalance = scanner.nextDouble();
                    scanner.nextLine();
                    account = new BankAccount(newName, newBalance); // reassign account variable to new object
                    System.out.println("New account created for " + newName + "!");
                    break;

                case 5: // EXIT
                    isRunning = false; // sets flag to false, loop exits naturally
                    break;

                default: // handles any input outside 1-5
                    System.out.println("Invalid choice, enter 1-5.");
            }
            System.out.println();
        }
        scanner.close(); // always close scanner to free resources
    }
}

// ============================================================
// BANKACCOUNT CLASS — owns all data and business logic
// This is where OOP concepts are demonstrated:
//   - Encapsulation: private fields, public methods
//   - Abstraction: Main doesn't know HOW methods work internally
//   - Constructor: initializes object with valid starting state
// ============================================================
class BankAccount {

    // ENCAPSULATION: fields are private
    // Nobody outside this class can directly read or change these
    private String ownerName;
    private double balance;

    // CONSTRUCTOR: called when 'new BankAccount(...)' is used
    // Sets up the object with initial values
    // 'this' is needed here because parameter names match field names
    // this.ownerName = the field | ownerName = the parameter
    BankAccount(String ownerName, double balance) {
        this.ownerName = ownerName;
        this.balance = balance;
    }

    // GETTER METHOD (accessor)
    // Since balance is private, this is the controlled way to read it
    // Also prints the balance — returns the value for use in calculations
    public double showBalance() {
        System.out.println("Balance of " + ownerName + " is " + balance);
        return this.balance; // returns value so caller can use it (e.g. case 2)
    }

    // DEPOSIT METHOD
    // Returns boolean so caller knows if it succeeded or failed
    // Validation: rejects zero or negative deposits
    public boolean deposit(double balance) {
        // 'balance' here is the parameter (amount to deposit)
        // 'this.balance' is the field (account's actual balance)
        // 'this' disambiguates them since they share the same name
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false; // early return — stops execution immediately
        }
        this.balance += balance; // shorthand for: this.balance = this.balance + balance
        return true;
    }

    // WITHDRAW METHOD
    // Two validations: invalid amount AND insufficient funds
    // Order matters: check invalid amount first, then check funds
    public boolean withdraw(double balance) {
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false;
        }
        if (balance > this.balance) { // can't withdraw more than available
            System.out.println("Insufficient funds. Balance: " + this.balance);
            return false;
        }
        this.balance -= balance; // deduct the amount
        return true;
    }
}

// ============================================================
// KEY OOP CONCEPTS DEMONSTRATED IN THIS PROGRAM:
//
// 1. ENCAPSULATION
//    - ownerName and balance are private
//    - Access only through public methods (deposit, withdraw, showBalance)
//    - Validation inside methods prevents invalid state (e.g. negative balance)
//
// 2. ABSTRACTION
//    - Main just calls account.deposit(500) without knowing how it works
//    - Complexity is hidden behind clean method names
//
// 3. CONSTRUCTOR
//    - Guarantees every BankAccount starts with valid owner and balance
//    - Called automatically when 'new BankAccount(...)' is used
//
// 4. 'this' KEYWORD
//    - Refers to the current object's fields
//    - Required when parameter name matches field name (constructor, deposit, withdraw)
//    - Not about private/public — about disambiguation
//
