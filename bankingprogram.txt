import java.util.Scanner;

class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        boolean isRunning = true;
        int choice = 0;

        System.out.print("Enter your name: ");
        String name = scanner.nextLine();
        System.out.print("Enter initial balance: ");
        double initialBalance = scanner.nextDouble();
        scanner.nextLine();

        // ===== OOP: Object creation (Class instantiation) =====
        BankAccount account;

        System.out.print("What type of account you want to open ? 1: Savings OR 2: Current");
        int accType = scanner.nextInt();

        // ===== OOP: Polymorphism (parent reference, child object) =====
        if(accType == 1) { 
            account = new savingsAccount(name, initialBalance); // Inheritance + Runtime polymorphism
        }
        else { 
            int overDraftLimit = 0;
            System.out.print("Enter the overdraft limit: ");
            overDraftLimit = scanner.nextInt();

            account = new currentAccount(name, initialBalance, overDraftLimit); // Inheritance + polymorphism
        }

        System.out.println("Account created! Welcome, " + name + "!\n");

        while (isRunning) {
            System.out.println("Banking Simulation");
            System.out.println();
            System.out.println("1. Deposit");
            System.out.println("2. Withdraw");
            System.out.println("3. Display");
            System.out.println("4. Open a new account");
            System.out.println("5. Exit");
            System.out.println();
            System.out.print("Enter choice 1-5: ");

            choice = scanner.nextInt();
            scanner.nextLine();

            switch (choice) {

                case 1:
                    System.out.print("Enter amount to deposit: ");
                    double deposit = scanner.nextDouble();
                    scanner.nextLine();

                    // ===== OOP: Method invocation on object =====
                    boolean res = account.deposit(deposit);
                    if (res) System.out.println("Deposited: " + deposit);
                    break;

                case 2:
                    System.out.print("Enter amount to withdraw: ");
                    double wit = scanner.nextDouble();
                    scanner.nextLine();

                    // ===== OOP: Runtime Polymorphism (method overridden in subclasses) =====
                    boolean withdrawn = account.withdraw(wit);

                    if (withdrawn) {
                        double remaining = account.showBalance();
                        System.out.println("Withdrawn: " + wit + " | Remaining: " + remaining);
                    }
                    break;

                case 3:
                    // ===== OOP: Encapsulation (accessing data via method, not directly) =====
                    account.showBalance();
                    break;

                case 4:
                    System.out.print("Enter new account holder name: ");
                    String newName = scanner.nextLine();
                    System.out.print("Enter initial balance: ");
                    double newBalance = scanner.nextDouble();
                    scanner.nextLine();

                    // NOTE: This breaks polymorphism design (direct base class usage)
                    account = new BankAccount(newName, newBalance);

                    System.out.println("New account created for " + newName + "!");
                    break;

                case 5:
                    isRunning = false;
                    break;

                default:
                    System.out.println("Invalid choice, enter 1-5.");
            }
            System.out.println();
        }
        scanner.close();
    }
}


// ===== OOP: Base Class =====
// Demonstrates Encapsulation + Abstraction (basic account behavior hidden behind methods)
class BankAccount {

    // ===== OOP: Encapsulation (data hiding using protected) =====
    protected String ownerName;
    protected double balance;

    BankAccount(String ownerName, double balance) {
        this.ownerName = ownerName;
        this.balance = balance;
    }

    // ===== OOP: Method abstraction (user interacts via method, not fields) =====
    public double showBalance() {
        System.out.println("Balance of " + ownerName + " is " + balance);
        return this.balance;
    }

    public boolean deposit(double balance) {
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false;
        }
        this.balance += balance;
        return true;
    }

    // ===== OOP: Method designed for overriding =====
    public boolean withdraw(double balance) {
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false;
        }
        if (balance > this.balance) {
            System.out.println("Insufficient funds. Balance: " + this.balance);
            return false;
        }
        this.balance -= balance;
        return true;
    }
}


// ===== OOP: Inheritance (IS-A relationship) =====
class savingsAccount extends BankAccount{
    
    protected float interest;
    
    savingsAccount(String ownerName, double balance){
        super(ownerName, balance); // ===== OOP: Constructor chaining =====
        this.interest = interest;
    }

    // ===== OOP: Method Overriding (Runtime Polymorphism) =====
    @Override
    public boolean withdraw(double balance) {
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false;
        }
        if (balance > this.balance) {
            System.out.println("Insufficient funds. Balance: " + this.balance);
            return false;
        }

        // Business rule override (Savings constraint)
        if (balance > 100) this.balance -= balance;
        else {
            System.out.println("Amount should be grater than 100");
            return false;
        }
        return true;
    }
}


// ===== OOP: Inheritance + Extension =====
class currentAccount extends BankAccount{

    private int overDraftLimit = 0;

    currentAccount(String ownerName, double balance, int overDraftLimit){
        super(ownerName, balance); // ===== Constructor chaining =====
        this.overDraftLimit = overDraftLimit;
    }

    // ===== OOP: Method Overriding (Runtime Polymorphism) =====
    @Override
    public boolean withdraw(double balance) {
        if (balance <= 0) {
            System.out.println("Enter a valid amount.");
            return false;
        }

        // Extended rule using overdraft (specialized behavior)
        if (balance > this.balance + this.overDraftLimit) {
            System.out.println("Insufficient funds. Balance: " + this.balance);
            return false;
        }
        this.balance -= balance;
        return true;
    }
}


/*
======================== OOP CONCEPTS USED (REVISION NOTES) ========================

1. CLASS
   Definition: Blueprint/template for creating objects.
   Example in code:
   - BankAccount, savingsAccount, currentAccount, Main

-----------------------------------------------------------------------------------

2. OBJECT (INSTANCE)
   Definition: A real-world entity created from a class using 'new'.
   Example in code:
   - account = new savingsAccount(...)
   - account = new currentAccount(...)
   - account = new BankAccount(...)

-----------------------------------------------------------------------------------

3. ENCAPSULATION
   Definition: Wrapping data (variables) + methods together and restricting direct access.
   Example in code:
   - protected String ownerName;
   - protected double balance;
   - Access done via methods like deposit(), withdraw(), showBalance()

-----------------------------------------------------------------------------------

4. INHERITANCE (IS-A RELATIONSHIP)
   Definition: One class acquires properties and behavior of another class using 'extends'.
   Example in code:
   - class savingsAccount extends BankAccount
   - class currentAccount extends BankAccount

-----------------------------------------------------------------------------------

5. POLYMORPHISM (RUNTIME POLYMORPHISM)
   Definition: Same method behaves differently based on object type at runtime.
   Example in code:
   - BankAccount account;
   - account = new savingsAccount(...)
   - account = new currentAccount(...)
   - account.withdraw(...) behaves differently for each class

-----------------------------------------------------------------------------------

6. METHOD OVERRIDING
   Definition: Child class provides its own implementation of parent method.
   Example in code:
   - savingsAccount overrides withdraw()
   - currentAccount overrides withdraw()

-----------------------------------------------------------------------------------

7. ABSTRACTION (PARTIAL)
   Definition: Hiding implementation details and showing only essential behavior.
   Example in code:
   - User calls account.withdraw(), deposit(), showBalance()
   - Internal logic differs but user doesn’t care

-----------------------------------------------------------------------------------

8. CONSTRUCTOR
   Definition: Special method used to initialize objects.
   Example in code:
   - BankAccount(String ownerName, double balance)
   - savingsAccount(...)
   - currentAccount(...)

-----------------------------------------------------------------------------------

9. SUPER KEYWORD
   Definition: Used to call parent class constructor.
   Example in code:
   - super(ownerName, balance);

-----------------------------------------------------------------------------------

10. METHOD OVERLOADING (NOT USED HERE)
   Definition: Same method name with different parameters in same class.

-----------------------------------------------------------------------------------

11. KEY OOP FLOW IN THIS PROGRAM
   - User chooses account type
   - Object created dynamically (savings or current)
   - Same reference (BankAccount) used
   - Different behavior executed (polymorphism)

===================================================================================
*/
